pipeline {
   agent any
    tools {
      maven 'Apache Maven 3.0.5'
    }
   stages {
    stage('git clone') {
		steps {
			git branch: 'main', credentialsId: 'github', url: 'https://github.com/damanika/Vehicles_Hub.git'
		}
	}
	stage('maven clean') {
		steps {
			sh 'mvn clean'
		}
	}
    stage('maven test') {
        steps {
           sh 'mvn test'
        }
    }
    stage('maven compile') {
        steps {
           sh 'mvn compile'
        }
    }
	stage('maven package') {
        steps {
           sh 'mvn package'
        }
	}
    }
}
